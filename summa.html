// Get the elements const hoverButton = document.getElementById('hoverButton'); const hiddenDiv = document.getElementById('hiddenDiv'); // Function to show the hidden div function showDiv() { hiddenDiv.style.display = 'block'; } // Function to hide the hidden div function hideDiv() { hiddenDiv.style.display = 'none'; } // Show the hidden div when hovering over the button hoverButton.addEventListener('mouseover', showDiv); // Keep the div visible if mouse is over it or the button hoverButton.addEventListener('mouseout', function(event) { // Check if the mouse is moving to the hiddenDiv if (!hiddenDiv.contains(event.relatedTarget)) { hideDiv(); } }); // Show the div if mouse enters the div itself hiddenDiv.addEventListener('mouseover', showDiv); // Hide the div if the mouse leaves the div hiddenDiv.addEventListener('mouseout', function(event) { // Check if the mouse is moving back to the button if (!hoverButton.contains(event.relatedTarget)) { hideDiv(); } });
